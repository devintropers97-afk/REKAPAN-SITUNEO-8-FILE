## 2. Export to CSV/Excel/PDF
```javascript
/**
 * File: /public/assets/js/export.js
 * Export Utilities
 */

const ExportHelper = {
    // Export to CSV
    exportToCSV: function(data, filename = 'export.csv') {
        let csv = '';
        
        // Header
        const headers = Object.keys(data[0]);
        csv += headers.join(',') + '\n';
        
        // Rows
        data.forEach(row => {
            const values = headers.map(header => {
                const value = row[header];
                return typeof value === 'string' && value.includes(',') 
                    ? `"${value}"` 
                    : value;
            });
            csv += values.join(',') + '\n';
        });
        
        this.downloadFile(csv, filename, 'text/csv');
    },
    
    // Export to Excel (using SheetJS)
    exportToExcel: function(data, filename = 'export.xlsx') {
        const ws = XLSX.utils.json_to_sheet(data);
        const wb = XLSX.utils.book_new();
        XLSX.utils.book_append_sheet(wb, ws, 'Sheet1');
        XLSX.writeFile(wb, filename);
    },
    
    // Export to PDF (using jsPDF)
    exportToPDF: function(elementId, filename = 'export.pdf') {
        const element = document.getElementById(elementId);
        
        html2pdf()
            .from(element)
            .set({
                margin: 10,
                filename: filename,
                html2canvas: { scale: 2 },
                jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }
            })
            .save();
    },
    
    // Helper: Download file
    downloadFile: function(content, filename, mimeType) {
        const blob = new Blob([content], { type: mimeType });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = filename;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
    }
};
```

## Task Application System
```php
<?php
/**
 * Task Application Controller
 */
class PartnerTaskController extends Controller {
    
    public function apply($taskId) {
        $task = Task::find($taskId);
        
        if (!$task || $task->status !== 'active') {
            return json(['error' => 'Task not found'], 404);
        }
        
        // Check if already applied
        $existing = TaskApplication::where('task_id', $taskId)
                                   ->where('partner_id', Auth::user()->partner->partner_id)
                                   ->first();
        
        if ($existing) {
            return json(['error' => 'You already applied'], 400);
        }
        
        // Check if task is full
        $currentWorkers = TaskApplication::where('task_id', $taskId)
                                        ->where('status', 'accepted')
                                        ->count();
        
        if ($currentWorkers >= $task->max_workers) {
            return json(['error' => 'Task is full'], 400);
        }
        
        DB::beginTransaction();
        
        try {
            // Create application
            $application = TaskApplication::create([
                'task_id' => $taskId,
                'partner_id' => Auth::user()->partner->partner_id,
                'status' => 'pending',
                'applied_at' => date('Y-m-d H:i:s')
            ]);
            
            // Notify admin
            Notification::create([
                'user_id' => 1,
                'type' => 'info',
                'title' => 'New Task Application',
                'message' => Auth::user()->full_name . ' applied for task: ' . $task->title,
                'action_url' => '/admin/tasks/' . $taskId . '/applications'
            ]);
            
            DB::commit();
            
            return json([
                'success' => true,
                'message' => 'Application submitted! Wait for admin approval.'
            ]);
            
        } catch (Exception $e) {
            DB::rollback();
            return json(['error' => 'Failed to apply'], 500);
        }
    }
    
    public function submit($taskId) {
        $application = TaskApplication::where('task_id', $taskId)
                                     ->where('partner_id', Auth::user()->partner->partner_id)
                                     ->where('status', 'accepted')
                                     ->first();
        
        if (!$application) {
            return json(['error' => 'Application not found'], 404);
        }
        
        // Validate file upload
        if (!isset($_FILES['work_file'])) {
            return json(['error' => 'Please upload your work'], 422);
        }
        
        DB::beginTransaction();
        
        try {
            // Upload file
            $file = $_FILES['work_file'];
            $fileName = 'task_' . $taskId . '_' . time() . '_' . $file['name'];
            $uploadPath = '/public/assets/uploads/tasks/' . $fileName;
            
            move_uploaded_file($file['tmp_name'], dirname(__DIR__, 3) . $uploadPath);
            
            // Update application
            $application->update([
                'work_file' => $uploadPath,
                'work_notes' => $_POST['notes'] ?? null,
                'submitted_at' => date('Y-m-d H:i:s'),
                'status' => 'submitted'
            ]);
            
            // Notify admin
            Notification::create([
                'user_id' => 1,
                'type' => 'success',
                'title' => 'Task Submitted',
                'message' => Auth::user()->full_name . ' submitted work for: ' . $application->task->title,
                'action_url' => '/admin/tasks/' . $taskId . '/submissions'
            ]);
            
            DB::commit();
            
            return json([
                'success' => true,
                'message' => 'Work submitted! Wait for admin review.'
            ]);
            
        } catch (Exception $e) {
            DB::rollback();
            return json(['error' => 'Failed to submit'], 500);
        }
    }
}
```

## Tanggungan Rules
```
JIKA CLIENT SEWA < 3 BULAN:
- Partner bertanggung jawab 85% (harus kembalikan komisi yang sudah dibayar)
- SPV bertanggung jawab 10% (jika ada)
- Manager bertanggung jawab 5% (jika ada)

Potongan dilakukan OTOMATIS dari saldo commission_balance mereka.

Example:
Client sewa Rp 150K/bulan √ó 2 bulan = Rp 300K total dibayar
Partner Tier 2 (40%) sudah dapat komisi: Rp 120K
SPV sudah dapat: Rp 30K
Manager sudah dapat: Rp 15K

Client STOP di bulan ke-2 (kurang dari 3 bulan).

Tanggungan:
- Partner: Rp 300K √ó 85% = Rp 255K (dipotong dari saldo)
- SPV: Rp 300K √ó 10% = Rp 30K (dipotong dari saldo)
- Manager: Rp 300K √ó 5% = Rp 15K (dipotong dari saldo)

Saldo mereka akan berkurang otomatis!
```

## Tanggungan Processing Code
```php
<?php
/**
 * Process Tanggungan when Client cancels <3 months
 */
function processTanggungan($orderId) {
    $order = Order::find($orderId);
    
    if ($order->order_type !== 'sewa') {
        throw new Exception('Tanggungan only for sewa orders');
    }
    
    // Calculate subscription duration
    $startDate = strtotime($order->subscription_start_date);
    $cancelDate = strtotime($order->cancelled_at);
    $monthsDuration = ($cancelDate - $startDate) / (30 * 24 * 60 * 60);
    
    if ($monthsDuration >= 3) {
        // No tanggungan if >= 3 months
        return [
            'tanggungan_applies' => false,
            'reason' => 'Subscription >= 3 months'
        ];
    }
    
    // Get total revenue paid
    $totalRevenue = Payment::where('order_id', $orderId)
                           ->where('status', 'verified')
                           ->sum('amount');
    
    $partner = Partner::find($order->partner_id);
    $spv = $partner && $partner->spv_id ? Spv::find($partner->spv_id) : null;
    $manager = $spv && $spv->manager_id ? Manager::find($spv->manager_id) : null;
    
    DB::beginTransaction();
    
    try {
        $tanggunganRecords = [];
        
        // Partner: 85%
        if ($partner) {
            $amount = $totalRevenue * 0.85;
            
            // Deduct from commission_balance
            Partner::find($partner->partner_id)->decrement('commission_balance', $amount);
            
            // Record tanggungan
            Tanggungan::create([
                'order_id' => $orderId,
                'partner_id' => $partner->partner_id,
                'role' => 'partner',
                'percentage' => 85,
                'amount' => $amount,
                'status' => 'processed',
                'processed_at' => date('Y-m-d H:i:s')
            ]);
            
            // Notify partner
            Notification::create([
                'user_id' => $partner->user_id,
                'type' => 'warning',
                'title' => 'Tanggungan Deduction',
                'message' => "Rp " . number_format($amount) . " deducted for order #{$order->order_number} (client < 3 months)"
            ]);
            
            $tanggunganRecords[] = [
                'role' => 'partner',
                'amount' => $amount
            ];
        }
        
        // SPV: 10%
        if ($spv) {
            $amount = $totalRevenue * 0.10;
            
            Spv::find($spv->spv_id)->decrement('commission_balance', $amount);
            
            Tanggungan::create([
                'order_id' => $orderId,
                'spv_id' => $spv->spv_id,
                'role' => 'spv',
                'percentage' => 10,
                'amount' => $amount,
                'status' => 'processed',
                'processed_at' => date('Y-m-d H:i:s')
            ]);
            
            Notification::create([
                'user_id' => $spv->user_id,
                'type' => 'warning',
                'title' => 'Tanggungan Deduction',
                'message' => "Rp " . number_format($amount) . " deducted (downline client < 3 months)"
            ]);
            
            $tanggunganRecords[] = [
                'role' => 'spv',
                'amount' => $amount
            ];
        }
        
        // Manager: 5%
        if ($manager) {
            $amount = $totalRevenue * 0.05;
            
            Manager::find($manager->manager_id)->decrement('commission_balance', $amount);
            
            Tanggungan::create([
                'order_id' => $orderId,
                'manager_id' => $manager->manager_id,
                'role' => 'manager',
                'percentage' => 5,
                'amount' => $amount,
                'status' => 'processed',
                'processed_at' => date('Y-m-d H:i:s')
            ]);
            
            Notification::create([
                'user_id' => $manager->user_id,
                'type' => 'warning',
                'title' => 'Tanggungan Deduction',
                'message' => "Rp " . number_format($amount) . " deducted (area client < 3 months)"
            ]);
            
            $tanggunganRecords[] = [
                'role' => 'manager',
                'amount' => $amount
            ];
        }
        
        DB::commit();
        
        return [
            'tanggungan_applies' => true,
            'total_revenue' => $totalRevenue,
            'months_duration' => round($monthsDuration, 2),
            'deductions' => $tanggunganRecords
        ];
        
    } catch (Exception $e) {
        DB::rollback();
        throw $e;
    }
}
```
